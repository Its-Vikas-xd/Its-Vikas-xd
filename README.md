<script>
document.addEventListener('DOMContentLoaded', function() {
  const texts = [
    "forge --ai-image-studio --danger-mode",
    "analyze --spotify-data --visualize",
    "transform --reality --with-code",
    "initiate --collaboration --now"
  ];
  let count = 0;
  let index = 0;
  let currentText = '';
  let letter = '';
  const typedTextSpan = document.querySelector('.typed-text');
  const cursorSpan = document.querySelector('.cursor');

  function type() {
    if (count === texts.length) {
      count = 0;
    }
    currentText = texts[count];
    letter = currentText.slice(0, ++index);
    typedTextSpan.textContent = letter;

    if (letter.length === currentText.length) {
      cursorSpan.style.display = 'inline-block'; // Ensure cursor is visible at end of typing
      setTimeout(() => {
        cursorSpan.style.display = 'none'; // Hide cursor during "deleting"
        erase();
      }, 2000); // Wait 2 seconds before erasing
    } else {
      setTimeout(type, 100);
    }
  }

  function erase() {
    if (index > 0) {
      letter = currentText.slice(0, --index);
      typedTextSpan.textContent = letter;
      setTimeout(erase, 50); // Erase faster
    } else {
      count++;
      cursorSpan.style.display = 'inline-block'; // Show cursor before next typing
      setTimeout(type, 500); // Wait a bit before typing next sentence
    }
  }

  type(); // Start the typing effect
});
